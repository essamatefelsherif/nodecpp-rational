446 r.den,
447 static_cast<I>(r.num / r.den),
448 static_cast<I>(r.num % r.den)
449 };
450
451 unsigned int reverse = 0u;

453 while (ts.r < zero){ ts.r += ts.d; --ts.q; }
454 while (rs.r < zero){ rs.r += rs.d; --rs.q; }

456 for( ;; ){

458 return reverse ? ts.q > rs.q : ts.q < rs.q;
